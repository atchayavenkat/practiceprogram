echo "Enter number"
read n
len=$(echo $n | wc -c)
len=$(( $len - 1 ))
echo "Your day $n in words : "
for (( i=1; i<=$len; i++ ))
do
day=$(echo $n | cut -c $i)

case $day in
echo-n "Monday" ;;
echo-n "Tuesday" ;;
echo-n "Wednesday";;
echo-n "Thursday" ;;
echo-n "Friday" ;;
echo-n "Saturday" ;;
echo-n "Sunday" ;;
esac
done
echo " "
